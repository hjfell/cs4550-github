More SQL/MySQl

WHERE, LIKE, NOT LIKE

mysql> SELECT first_name, last_name FROM users WHERE email NOT LIKE '%@beatles.com';
mysql> SELECT * FROM users WHERE last_name LIKE '%own';

ORDER BY

mysql> SELECT * FROM users ORDER BY registration_date;
mysql> SELECT * FROM users ORDER BY last_name, first_name;
mysql> SELECT first_name, last_name FROM users WHERE last_name="Simpson" ORDER BY first_name;
mysql> SELECT first_name, last_name FROM users WHERE last_name!="Simpson" ORDER BY last_name;

LIMIT

mysql> SELECT * FROM users LIMIT 4;
mysql> SELECT * FROM users LIMIT 4, 5;
mysql> SELECT first_name, last_name FROM users ORDER BY registration_date DESC LIMIT 5;

UPDATE

mysql> INSERT into users (first_name, last_name, email, pass, registration_date)
    -> VALUES ('Michael','Chabon','m.chabon@alaska.com',SHA1('verycold'),NOW());
mysql> SELECT user_id FROM users WHERE first_name="Michael" AND last_name="Chabon";
mysql> UPDATE users
    -> SET email='mike@authors.com' WHERE user_id = 14;
mysql> SELECT * FROM users WHERE user_id = 14;

Some Functions

mysql> SELECT UPPER(first_name) FROM users ORDER by last_name;
mysql> SELECT CONCAT(last_name, ', ', first_name)
    -> AS Name FROM users ORDER BY Name;
mysql> SELECT RAND();
mysql> SELECT * FROM users ORDER BY RAND();
mysql> SELECT email FROM users ORDER BY RAND() LIMIT 1;
mysql> SELECT DATE(registration_date) FROM users;
mysql> SELECT DAYNAME(registration_date) FROM users;
